select sum(l_extendedprice * (1 - l_discount)/1000) from ids as id1, ids as id2, nation as n1, nation as n2, customer, orders, lineitem, supplier where n1.N_NATIONKEY=customer.C_NATIONKEY and n2.N_NATIONKEY=supplier.S_NATIONKEY and customer.C_CUSTKEY=orders.O_CUSTKEY and orders.O_ORDERKEY=lineitem.L_ORDERKEY and lineitem.L_SUPPKEY=supplier.S_SUPPKEY  and id1.i_id = supplier.s_id and id2.i_id = customer.c_id;
